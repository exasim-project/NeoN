117template<typename ValueType, typename IndexType, typename SparsityType>
118LinearSystem<ValueType, IndexType> createEmptyLinearSystem(const SparsityType& sparsity)
119{
120 const auto& exec = sparsity.mesh().exec();
121
122 localIdx rows {sparsity.rows()};
123 localIdx nnzs {sparsity.nnz()};
124
125 return {
126 CSRMatrix<ValueType, IndexType> {
127 Vector<ValueType>(exec, nnzs, zero<ValueType>()), sparsity.colIdxs(), sparsity.rowOffs()
128 },
129 Vector<ValueType> {exec, rows, zero<ValueType>()}
130 };
131}
