133/*@brief convenience function to call log on logger with std::format */
134inline void log(std::shared_ptr<BaseLogger> logger, LogEvent& event, std::string_view delim = ",")
135{
136 if (logger != nullptr)
137 {
138 // console output
139 auto formattedMessage =
140 logger->getTarget() == Target::Console ? event.message : event.toJson(delim);

142 logger->log(formattedMessage);
