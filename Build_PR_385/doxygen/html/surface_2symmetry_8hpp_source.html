19inline void applySymmetry(
20 Field<scalar>& domainVector, const UnstructuredMesh& mesh, std::pair<size_t, size_t> range
21)
22{
23 auto refValue = domainVector.boundaryData().refValue().view();
24 auto value = domainVector.boundaryData().value().view();

26

43// Vec3 specialization ----------------------------------------------------------
44inline void applySymmetry(
45 Field<Vec3>& domainVector, const UnstructuredMesh& mesh, std::pair<size_t, size_t> range
46)
47{
48 auto refValue = domainVector.boundaryData().refValue().view();
49 auto value = domainVector.boundaryData().value().view();
50 auto internalValues = domainVector.internalVector().view();
51
52 auto faceCells = mesh.boundaryMesh().faceCells().view();
53 auto nHat = mesh.boundaryMesh().nf().view(); // unit normals
54 const localIdx nInternalFaces = mesh.nInternalFaces();
55
56 NeoN::parallelFor(
57 domainVector.exec(),
58 range,
59 KOKKOS_LAMBDA(const localIdx i) {
60 const localIdx owner = faceCells[i];
61 const Vec3 n = nHat[i];
62
63 Vec3 v = internalValues[owner];
64 const scalar vn = v & n; // dot product
65 const auto vtan = v - n * vn;
66
67 refValue[i] = vtan;
68 value[i] = vtan;
69 }
70 );
71}
