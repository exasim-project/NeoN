59 SurfScalarField ddtPhiCorr(const VolVectorField& U, const SurfScalarField& phi, scalar dt) const
60 {
61 const auto& mesh = U.mesh();
62 const auto exec = phi.exec();
63
64 // --- interpolation (same for all schemes)
65 NeoN::finiteVolume::cellCentred::SurfaceInterpolation<Vec3> interp(
66 exec, mesh, NeoN::TokenList({std::string("linear")})
67 );
68
69 // --- boundary conditions consistent with U
70 auto surfaceBCs = NeoN::finiteVolume::cellCentred::createPhiCorrBCsFromU(mesh, U);
71
72 SurfScalarField phiCorr(exec, std::string("ddtPhiCorr"), mesh, surfaceBCs);
73
74 auto [outV, SfV] = views(phiCorr.internalVector(), mesh.faceAreas());
75
76 const size_t N = outV.size();
77
78 // ===============================
79 // BDF1 contribution (phi^{n})
80 // ===============================
81 const auto& U0 = NeoN::finiteVolume::cellCentred::oldTime(U);
82 const auto& phi0 = NeoN::finiteVolume::cellCentred::oldTime(phi);
83
84 auto Uf0 = interp.interpolate(U0);
85
86 auto [phi0V, Uf0V] = views(phi0.internalVector(), Uf0.internalVector());
87
88 const scalar w1 = a1(dt);
89
90 NeoN::parallelFor(
91 exec,
92 {size_t(0), N},
93 KOKKOS_LAMBDA(const localIdx i) {
94 const auto d = (SfV[i] & Uf0V[i]);
95 const auto corr = phi0V[i] - d;
96
97 const scalar limiter = ddtPhiCorrLimiter(mag(phi0V[i]), mag(corr));
98
99 outV[i] += limiter * w1 * corr;
100 },
101 "ddtPhiCorr::BDF1"
102 );
103
104 // ===============================
105 // BDF2 contribution (phi^{n-1})
106 // ===============================
107 if (nSteps() >= 2)
108 {
109 const auto& U00 =
110 NeoN::finiteVolume::cellCentred::oldTime(NeoN::finiteVolume::cellCentred::oldTime(U)
111 );
112 const auto& phi00 = NeoN::finiteVolume::cellCentred::oldTime(
113 NeoN::finiteVolume::cellCentred::oldTime(phi)
114 );
115
116 auto Uf00 = interp.interpolate(U00);
117
118 auto [phi00V, Uf00V] = views(phi00.internalVector(), Uf00.internalVector());
119
120 const scalar w2 = a2(dt);
121
122 NeoN::parallelFor(
123 exec,
124 {size_t(0), N},
125 KOKKOS_LAMBDA(const localIdx i) {
126 const auto d = (SfV[i] & Uf00V[i]);
127 const auto corr = phi00V[i] - d;
128
129 const scalar limiter = ddtPhiCorrLimiter(mag(phi00V[i]), mag(corr));
130
131 outV[i] += limiter * w2 * corr;
132 },
133 "ddtPhiCorr::BDF2"
134 );
135 }
136
137 phiCorr.correctBoundaryConditions();
138 return phiCorr;
139 }
140
141protected:
142
143 // ------------------------------------------------------------------ //
144 // OpenFOAM-compatible limiter (with numerical safety)
145 // ------------------------------------------------------------------ //
146 KOKKOS_INLINE_FUNCTION
147 static scalar ddtPhiCorrLimiter(const scalar phiMag, const scalar corrMag)
148 {
149 // OpenFOAM SMALL (double precision)
150 constexpr scalar SMALL = scalar(1e-30);
151
152 const scalar ratio = corrMag / (phiMag + SMALL);
153 return scalar(1.0) - Kokkos::min(ratio, scalar(1.0));
154 }
155};
