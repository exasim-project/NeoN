28 // -------- scalar --------
29 SolverStats solve(const LinearSystem<scalar, localIdx>& sys, Vector<scalar>& x) const override
30 {
31 auto start = std::chrono::steady_clock::now();
32
33 const auto rhs = sys.rhs();
34 const auto mtx = sys.matrix();
35 const auto mtxV = mtx.view();
36
37 auto [xV, bV] = views(x, rhs);
38
39 parallelFor(
40 x.exec(),
41 {0, bV.size()},
42 KOKKOS_LAMBDA(const localIdx i) {
43 const auto rowBegin = mtxV.rowOffs[i];
44 const auto rowEnd = mtxV.rowOffs[i + 1];
45
46 localIdx diagIdx = -1;
47 for (localIdx k = rowBegin; k < rowEnd; ++k)
48 {
49 if (mtxV.colIdxs[k] == i)
50 {
51 diagIdx = k;
52 break;
53 }
54 }
55
56 if (diagIdx < 0)
57 {
58 Kokkos::abort("DiagonalSolver: diagonal entry not found");
59 }
60
61 xV[i] = bV[i] / mtxV.values[diagIdx];
62 },
63 "DiagonalSolver::solve<scalar>"
64 );
65
66 fence(x.exec());
67
68 auto end = std::chrono::steady_clock::now();
69 scalar ms =
70 std::chrono::duration_cast<std::chrono::microseconds>(end - start).count() / 1000.0;
71
72 return {1, 0.0, 0.0, ms};
73 }
