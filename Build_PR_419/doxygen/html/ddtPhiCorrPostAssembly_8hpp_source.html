24inline void applyDivToRhs(
25 const fvcc::SurfaceField<scalar>& phiCorr, NeoN::la::LinearSystem<scalar, NeoN::localIdx>& ls
26)
27{
28 const auto& mesh = phiCorr.mesh();
29 const auto exec = phiCorr.exec();
30
31 auto [rhs, owner, neighbour, phiV] =
32 views(ls.rhs(), mesh.faceOwner(), mesh.faceNeighbour(), phiCorr.internalVector());
33
34 const auto nInternalFaces = mesh.nInternalFaces();
35
36 NeoN::parallelFor(
37 exec,
38 {NeoN::localIdx(0), NeoN::localIdx(nInternalFaces)},
39 KOKKOS_LAMBDA(const NeoN::localIdx f) {
40 const auto flux = phiV[f];
41 rhs[owner[f]] -= flux;
42 rhs[neighbour[f]] += flux;
43 },
44 "postAssembly::DdtPhiCorr"
45 );
46}
