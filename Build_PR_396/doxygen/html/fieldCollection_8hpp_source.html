391 std::int64_t iterationIndex = std::numeric_limits<std::int64_t>::max();
392 std::int64_t subCycleIndex = std::numeric_limits<std::int64_t>::max();
393
394 VectorDocument operator()(Database& db)
395 {
396 Field<typename VectorType::VectorValueType> domainVector(
397 field.mesh().exec(), field.internalVector(), field.boundaryData()
398 );
399
400 VectorType vf(
401 field.exec(), name, field.mesh(), domainVector, field.boundaryConditions(), db, "", ""
402 );
403
404 if (field.registered())
405 {
406 const VectorCollection& fieldCollection = VectorCollection::instance(field);
407 const VectorDocument& fieldDoc = fieldCollection.fieldDoc(field.key);
408 if (timeIndex == std::numeric_limits<std::int64_t>::max())
409 {
410 timeIndex = fieldDoc.timeIndex();
411 }
412 if (iterationIndex == std::numeric_limits<std::int64_t>::max())
413 {
414 iterationIndex = fieldDoc.iterationIndex();
415 }
416 if (subCycleIndex == std::numeric_limits<std::int64_t>::max())
417 {
418 subCycleIndex = fieldDoc.subCycleIndex();
419 }
420 }
421 return NeoN::Document(
422 {{"name", vf.name},
423 {"timeIndex", timeIndex},
424 {"iterationIndex", iterationIndex},
425 {"subCycleIndex", subCycleIndex},
426 {"field", vf}},
