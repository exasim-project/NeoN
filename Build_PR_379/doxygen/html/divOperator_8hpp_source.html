92 // copy constructor

98 divOperatorStrategy_(
99 divOp.divOperatorStrategy_ ? divOp.divOperatorStrategy_->clone() : nullptr

102 DivOperator(

104 const SurfaceField<scalar>& faceFlux,
105 const VolumeField<ValueType>& phi,
106 Input input
107 )
108 : dsl::OperatorMixin<VolumeField<ValueType>>(phi.exec(), dsl::Coeff(1.0), phi, termType),
109 faceFlux_(faceFlux),
110 divOperatorStrategy_(DivOperatorFactory<ValueType>::create(phi.exec(), phi.mesh(), input)
111 ) {};
112
113 DivOperator(
114 dsl::Operator::Type termType,
115 const SurfaceField<scalar>& faceFlux,
116 const VolumeField<ValueType>& phi,
117 std::unique_ptr<DivOperatorFactory<ValueType>> divOperatorStrategy
118 )
119 : dsl::OperatorMixin<VolumeField<scalar>>(phi.exec(), dsl::Coeff(1.0), phi, termType),
120 faceFlux_(faceFlux), divOperatorStrategy_(std::move(divOperatorStrategy)) {};
121
122 DivOperator(
123 dsl::Operator::Type termType,
124 const SurfaceField<scalar>& faceFlux,
125 const VolumeField<ValueType>& phi
126 )
127 : dsl::OperatorMixin<VolumeField<ValueType>>(phi.exec(), dsl::Coeff(1.0), phi, termType),
128 faceFlux_(faceFlux), divOperatorStrategy_(nullptr) {};
129
130
131 void explicitOperation(Vector<ValueType>& source) const
132 {
133 NF_ASSERT(divOperatorStrategy_, "DivOperatorStrategy not initialized");
134 auto tmpsource = Vector<ValueType>(source.exec(), source.size(), zero<ValueType>());
135 const auto operatorScaling = this->getCoefficient();
136 divOperatorStrategy_->div(tmpsource, faceFlux_, this->getVector(), operatorScaling);
137 source += tmpsource;
138 }
139
140 la::LinearSystem<ValueType, localIdx> createEmptyLinearSystem() const
141 {
142 NF_ASSERT(divOperatorStrategy_, "DivOperatorStrategy not initialized");
143 return divOperatorStrategy_->createEmptyLinearSystem();
144 }
145
146 void implicitOperation(la::LinearSystem<ValueType, localIdx>& ls) const
147 {
148 NF_ASSERT(divOperatorStrategy_, "DivOperatorStrategy not initialized");
149 const auto operatorScaling = this->getCoefficient();
150 divOperatorStrategy_->div(ls, faceFlux_, this->getVector(), operatorScaling);
151 }
152
153 [[deprecated("use explicit or implicit operation")]] void div(auto&&... args) const
154 {
155 const auto operatorScaling = this->getCoefficient();
156 divOperatorStrategy_->div(
157 std::forward<decltype(args)>(args)..., faceFlux_, this->getVector(), operatorScaling
158 );
159 }
160
161 void read(const Input& input)
162 {
163 const UnstructuredMesh& mesh = this->getVector().mesh();
164 if (std::holds_alternative<NeoN::Dictionary>(input))
165 {
166 auto dict = std::get<NeoN::Dictionary>(input);
167 std::string schemeName = "div(" + faceFlux_.name + "," + this->getVector().name + ")";
168 auto tokens = dict.subDict("divSchemes").get<NeoN::TokenList>(schemeName);
169 divOperatorStrategy_ =
170 DivOperatorFactory<ValueType>::create(this->exec(), mesh, tokens);
171 }
172 else
173 {
174 auto tokens = std::get<NeoN::TokenList>(input);
175 divOperatorStrategy_ =
176 DivOperatorFactory<ValueType>::create(this->exec(), mesh, tokens);
177 }
178 }
179
180 std::string getName() const { return "DivOperator"; }
181
182private:
183
184 const SurfaceField<NeoN::scalar>& faceFlux_;
185
186 std::unique_ptr<DivOperatorFactory<ValueType>> divOperatorStrategy_;
187};
