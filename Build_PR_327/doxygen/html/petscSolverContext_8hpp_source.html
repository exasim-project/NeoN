71 //- Create auxiliary rows for calculation purposes
72 void initialize(const LinearSystem<scalar, localIdx>& sys)
73 {
74
75 setOption(solverDict_);
76
77 std::size_t size = sys.matrix().values().size();
78 std::size_t nrows = sys.rhs().size();
79 PetscInt colIdx[size];
80 PetscInt rowIdx[size];
81 PetscInt rhsIdx[nrows];
82
83 // colIdx = sys.matrix().colIdxs().data();
84
85 auto hostLS = sys.copyToHost();
86
87 // auto fieldS = field.view();
88
89 auto rowPtrHost = hostLS.matrix().rowOffs().view();
90 auto colIdxHost = hostLS.matrix().colIdxs().view();
91 auto rhsHost = sys.rhs().copyToHost();
92
93 for (size_t index = 0; index < nrows; ++index)
94 {
95 rhsIdx[index] = static_cast<PetscInt>(index);
96 }
97 // copy colidx
98 // TODO: (this should be done only once when the matrix
99 // topology changes
100 for (size_t index = 0; index < size; ++index)
101 {
102 colIdx[index] = static_cast<PetscInt>(colIdxHost[index]);
103 }
104 // convert rowPtr to rowIdx
105 // TODO: (this should be done only once when the matrix
106 // topology changes
107 size_t rowI = 0;
108 size_t rowOffset = rowPtrHost[rowI + 1];
109 for (size_t index = 0; index < size; ++index)
110 {
111 if (index == rowOffset)
112 {
113 rowI++;
114 rowOffset = rowPtrHost[rowI + 1];
115 }
116 rowIdx[index] = rowI;
117 }
118
119 // move all not necessary staff to outer most scope since matrix has
120 // to be preallocated only once every time the mesh changes
121 PetscInitialize(NULL, NULL, 0, NULL);
122
123 MatCreate(PETSC_COMM_WORLD, &Amat_);
124 MatSetSizes(Amat_, sys.matrix().nRows(), sys.rhs().size(), PETSC_DECIDE, PETSC_DECIDE);
125
126 VecCreate(PETSC_COMM_SELF, &rhs_);
127 VecSetSizes(rhs_, PETSC_DECIDE, nrows);
128
129 std::string execName = std::visit([](const auto& e) { return e.name(); }, exec_);
130
131 if (execName == "GPUExecutor")
132 {
133 VecSetType(rhs_, VECKOKKOS);
134 MatSetType(Amat_, MATAIJKOKKOS);
135 }
136 else
137 {
138 VecSetType(rhs_, VECSEQ);
139 MatSetType(Amat_, MATSEQAIJ);
140 }
141 VecDuplicate(rhs_, &sol_);
142
143 VecSetPreallocationCOO(rhs_, nrows, rhsIdx);
144 MatSetPreallocationCOO(Amat_, size, colIdx, rowIdx);
145
146 KSPCreate(PETSC_COMM_WORLD, &ksp_);
147 KSPSetFromOptions(ksp_);
148 KSPSetOperators(ksp_, Amat_, Amat_);
149
150
151 init_ = true;
152
153 // PetscOptions options;
154 // PetscOptionsCreate(&options);
155 // PetscOptionsSetValue(NULL, "-no_signal_handler", "true");
156 PetscOptionsView(NULL, PETSC_VIEWER_STDOUT_WORLD);
157 }

167 for (auto key : solverDict_.subDict("options").keys())
168 {
169
170 std::string petscOptionKey = std::string("-") + key;
171 std::string petscOptionVal = subDict.get<std::string>(key);
172 PetscOptionsSetValue(NULL, petscOptionKey.c_str(), petscOptionVal.c_str());
173 }
