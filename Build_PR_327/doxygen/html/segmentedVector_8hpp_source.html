24template<typename IndexType>
25IndexType segmentsFromIntervals(const Vector<IndexType>& intervals, Vector<IndexType>& offsets)
26{
27 IndexType finalValue = 0;
28 const auto inView = intervals.view();
29 // skip the first element of the offsets
30 // assumed to be zero
31 auto offsView = offsets.view().subview(1);
32 // NOTE avoid compiler warning by static_casting to localIdx since offsView
33 // is a View
34 NF_ASSERT_EQUAL(inView.size(), offsView.size());
35 NeoN::parallelScan(
36 intervals.exec(),
37 {0, offsView.size()},
38 KOKKOS_LAMBDA(const localIdx i, IndexType& update, const bool final) {
39 update += inView[i];
40 if (final)
41 {
42 // offView is a view, thus [] takes unsigned idx
43 offsView[i] = update;
44 }
45 },
46 finalValue
47 );
48 return finalValue;
49}
