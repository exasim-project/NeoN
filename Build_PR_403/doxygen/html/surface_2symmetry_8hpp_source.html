45// Vec3 specialization ----------------------------------------------------------
46inline void applySymmetry(
47 Field<Vec3>& domainVector, const UnstructuredMesh& mesh, std::pair<size_t, size_t> range
48)
49{
50 auto [refValueV, valueV, internalValuesV, faceCellsV, nHatV] = views(
51 domainVector.boundaryData().refValue(),
52 domainVector.boundaryData().value(),
53 domainVector.internalVector(),
54 mesh.boundaryMesh().faceCells(),
55 mesh.boundaryMesh().nf()
56 );
57
58 const localIdx nInternalFaces = mesh.nInternalFaces();
59
60 NeoN::parallelFor(
61 domainVector.exec(),
62 range,
63 KOKKOS_LAMBDA(const localIdx i) {
64 const localIdx owner = faceCellsV[i];
65 const Vec3 n = nHatV[i];
66
67 Vec3 v = internalValuesV[owner];
68 const scalar vn = v & n; // dot product
69 const auto vtan = v - n * vn;
70
71 refValueV[i] = vtan;
72 valueV[i] = vtan;
73 }
74 );
75}
