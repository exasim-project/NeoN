16template<typename ValueType>
17[[nodiscard]] la::LinearSystem<ValueType, localIdx> assembleLinearSystem(
18 dsl::Expression<ValueType> expr,
19 VolumeField<ValueType>& rhs,
20 const Dictionary& fvSchemes,
21 scalar t,
22 scalar dt
23)
24{
25 auto sp = SparsityPattern(rhs.mesh());
26 auto exec = expr.exec();
27
28 la::LinearSystem<ValueType, localIdx> ls =
29 la::createEmptyLinearSystem<ValueType, localIdx, SparsityPattern>(sp);
30
31 auto nCells = rhs.size();
32 Vector<ValueType> source(exec, nCells);
33
34 fill(ls.rhs(), zero<ValueType>());
35 fill(ls.matrix().values(), zero<ValueType>());
36
37 auto [csrHV, rhsHV] = ls.view();
38
39 expr.explicitOperation(source);
40 expr.explicitOperation(source, t, dt);
41 expr.implicitOperation(ls);
42 expr.implicitOperation(ls, t, dt);
43
44 auto [rhsV, sourceV] = views(ls.rhs(), source);
45
46 // subtract the explicit source term from the rhs
47 auto vol = rhs.mesh().cellVolumes().view();
48 NeoN::parallelFor(
49 exec, {0, rhs.size()}, KOKKOS_LAMBDA(const localIdx i) { rhsV[i] -= sourceV[i] * vol[i]; }
50 );
51
52 return ls;
53}
