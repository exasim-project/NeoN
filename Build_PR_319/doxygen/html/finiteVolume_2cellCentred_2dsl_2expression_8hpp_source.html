80 void assemble(scalar t, scalar dt)
81 {
82 auto vol = psi_.mesh().cellVolumes().view();
83 auto expSource = expr_.explicitOperation(psi_.mesh().nCells());
84 expr_.explicitOperation(expSource, t, dt);
85 auto expSourceView = expSource.view();
86 fill(ls_.rhs(), zero<ValueType>());
87 fill(ls_.matrix().values(), zero<ValueType>());
88 expr_.implicitOperation(ls_);
89 // TODO rename implicitOperation -> assembleLinearSystem
90 expr_.implicitOperation(ls_, t, dt);
91 auto rhs = ls_.rhs().view();
92 // we subtract the explicit source term from the rhs
93 NeoN::parallelFor(
94 exec(),
95 {0, rhs.size()},
96 KOKKOS_LAMBDA(const localIdx i) { rhs[i] -= expSourceView[i] * vol[i]; }
97 );
98 }

159 void setReference(const IndexType refCell, ValueType refValue)
160 {
161 // TODO currently assumes that matrix is already assembled
162 const auto diagOffset = sparsityPattern_->diagOffset().view();
163 const auto rowOffs = ls_.matrix().rowOffs().view();
164 auto rhs = ls_.rhs().view();
165 auto values = ls_.matrix().values().view();
166 NeoN::parallelFor(
167 ls_.exec(),
168 {refCell, refCell + 1},
169 KOKKOS_LAMBDA(const std::size_t refCelli) {
170 auto diagIdx = rowOffs[refCelli] + diagOffset[refCelli];
171 auto diagValue = values[diagIdx];
172 rhs[refCelli] += diagValue * refValue;
173 values[diagIdx] += diagValue;
174 }
175 );
176 }
