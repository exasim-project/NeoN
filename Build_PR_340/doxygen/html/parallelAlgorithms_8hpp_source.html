241template<typename Executor, typename ValueType>
242void parallelSort(
243 [[maybe_unused]] const Executor& exec, std::pair<localIdx, localIdx> range, ValueType* ptr
244)
245{
246 auto [start, end] = range;
247 ValueType* begin = ptr + start;
248
249 using runOn = typename Executor::exec;
250 using TeamPol = Kokkos::TeamPolicy<runOn>;
251 using TeamMem = typename TeamPol::member_type;
252
253 int vectorLen = TeamPol::vector_length_max();
254 Kokkos::parallel_for(
255 TeamPol(1, Kokkos::AUTO(), vectorLen),
256 KOKKOS_LAMBDA(const TeamMem& t) {
257 auto view = Kokkos::View<ValueType*, runOn, Kokkos::MemoryTraits<Kokkos::Unmanaged>>(
258 begin, end - start
259 );
260 Kokkos::Experimental::sort_team(t, view);
261 }
262 );
263}
