368template<typename VectorType>
369class CreateFromExistingVector
370{
371public:
372
373 std::string name;
374 const VectorType& field;
375 std::int64_t timeIndex = std::numeric_limits<std::int64_t>::max();
376 std::int64_t iterationIndex = std::numeric_limits<std::int64_t>::max();
377 std::int64_t subCycleIndex = std::numeric_limits<std::int64_t>::max();
378
379 VectorDocument operator()(Database& db)
380 {
381 Field<typename VectorType::VectorValueType> domainVector(
382 field.mesh().exec(), field.internalVector(), field.boundaryData()
383 );
384
385 VectorType vf(
386 field.exec(), name, field.mesh(), domainVector, field.boundaryConditions(), db, "", ""
387 );
388
389 if (field.registered())
390 {
391 const VectorCollection& fieldCollection = VectorCollection::instance(field);
392 const VectorDocument& fieldDoc = fieldCollection.fieldDoc(field.key);
393 if (timeIndex == std::numeric_limits<std::int64_t>::max())
394 {
395 timeIndex = fieldDoc.timeIndex();
396 }
397 if (iterationIndex == std::numeric_limits<std::int64_t>::max())
398 {
399 iterationIndex = fieldDoc.iterationIndex();
400 }
401 if (subCycleIndex == std::numeric_limits<std::int64_t>::max())
402 {
403 subCycleIndex = fieldDoc.subCycleIndex();
404 }
405 }
406 return NeoN::Document(
407 {{"name", vf.name},
408 {"timeIndex", timeIndex},
409 {"iterationIndex", iterationIndex},
410 {"subCycleIndex", subCycleIndex},
411 {"field", vf}},
412 validateVectorDoc
413 );
414 }
415};
