18/* @class Factory class to create time integration method by a given name
19 * using NeoNs runTimeFactory mechanism
20 */
21template<typename SolutionType>
22class TimeIntegratorBase :
23 public RuntimeSelectionFactory<
24 TimeIntegratorBase<SolutionType>,
25 Parameters<const Dictionary&, const Dictionary&>>
26{
27
28public:
29
30 using ValueType = typename SolutionType::VectorValueType;
31 using Expression = NeoN::dsl::Expression<ValueType>;
32 using VolVectorField = NeoN::finiteVolume::cellCentred::VolumeField<Vec3>;
33 using SurfScalarField = NeoN::finiteVolume::cellCentred::SurfaceField<scalar>;
34
35 static std::string name() { return "timeIntegrationFactory"; }
36
37 TimeIntegratorBase(const Dictionary& schemeDict, const Dictionary& solutionDict)
38 : schemeDict_(schemeDict), solutionDict_(solutionDict)
39 {}
40
41 virtual ~TimeIntegratorBase() {}
42
43 virtual void solve(
44 Expression& eqn, SolutionType& sol, scalar t, scalar dt
45 ) = 0; // Pure virtual function for solving
46
47 // Pure virtual function for cloning
48 virtual std::unique_ptr<TimeIntegratorBase> clone() const = 0;
49
50 virtual bool explicitIntegration() const { return true; }
51
52 virtual SurfScalarField
53 ddtPhiCorr(const VolVectorField& U, const SurfScalarField& phi, scalar dt) const
54 {
55 // construct a surface field with the same mesh & default BCs
56 auto surfaceBCs = NeoN::finiteVolume::cellCentred::createCalculatedBCs<
57 NeoN::finiteVolume::cellCentred::SurfaceBoundary<scalar>>(phi.mesh());
58 SurfScalarField out(phi.exec(), std::string("ddtPhiCorr"), phi.mesh(), surfaceBCs);
59
60 // fill internal & boundary with zero
61 NeoN::fill(out.internalVector(), scalar(0));
62 NeoN::fill(out.boundaryData().value(), scalar(0));
63 NeoN::fill(out.boundaryData().refValue(), scalar(0));
64 NF_INFO("Warning! ddtPhiCorr not implemented for this ddtScheme."
65 "Falling back to phiHbyA = flux(HbyA).");
66 return out;
67 }
68
69protected:
70
71 const Dictionary& schemeDict_;
72 const Dictionary& solutionDict_;
73};

100 schemeDict.get<std::string>("type"), schemeDict, solutionDict
101 )) {};
102
103 void solve(Expression& eqn, SolutionVectorType& sol, scalar t, scalar dt)
104 {
105 timeIntegratorStrategy_->solve(eqn, sol, t, dt);

113 }
114
115private:
