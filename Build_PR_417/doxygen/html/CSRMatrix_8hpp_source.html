49 KOKKOS_INLINE_FUNCTION
50 ValueType& entry(const IndexType i, const IndexType j) const
51 {
52 const IndexType rowSize = rowOffs[i + 1] - rowOffs[i];
53 for (std::remove_const_t<IndexType> ic = 0; ic < rowSize; ++ic)
54 {
55 const IndexType localCol = rowOffs[i] + ic;
56 if (colIdxs[localCol] == j)
57 {
58 return values[localCol];
59 }
60 if (colIdxs[localCol] > j) break;
61 }
62 Kokkos::abort("Memory not allocated for CSR matrix component.");

64 }
